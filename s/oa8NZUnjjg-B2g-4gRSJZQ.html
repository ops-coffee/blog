иҮӘе®ҡд№үshape

дёәдәҶзј–иҫ‘ж–№дҫҝпјҢжҲ‘жғіеңЁpaletteдёҠж·»еҠ дёҖдёӘshapeиҜҘеҰӮдҪ•ж“ҚдҪңе‘ўпјҹиҝҷйҮҢжҲ‘д»¬йңҖиҰҒиҮӘе®ҡд№үPaletteпјҢиҮӘе®ҡд№үPaletteжңүдёӨз§Қж–№ејҸеҸҜд»ҘйҖүжӢ©пјҢз¬¬дёҖз§Қе°ұжҳҜеҹәдәҺй»ҳи®Өзҡ„PaletteжқҘдҝ®ж”№пјҢз¬¬дәҢз§Қе°ұжҳҜе®Ңе…ЁеҶҷдёӘж–°зҡ„PaletteжқҘжӣҝд»Јй»ҳи®Өзҡ„PaletteпјҢз¬¬дёҖз§ҚеҸӘиғҪеңЁй»ҳи®Өзҡ„PaletteдёҠж·»еҠ shapeпјҢиҖҢдёҚиғҪдҝ®ж”№жҲ–еҲ йҷӨпјҢжҜ”иҫғйёЎиӮӢжҲ‘д»¬е°ұзӣҙжҺҘж”ҫејғдәҶпјҢжқҘзңӢдёӢе®Ңе…ЁиҮӘе®ҡд№үPaletteиҜҘеҰӮдҪ•е®һзҺ°

д»ҘдёӢд»Јз ҒеҹәдәҺжҲ‘д»¬д№ӢеүҚжҗӯе»әеҘҪзҡ„д»Јз ҒжЎҶжһ¶пјҢе…·дҪ“еҸҜзңӢж–Үз« гҖҺжңҖеҘҪз”Ёзҡ„жөҒзЁӢзј–иҫ‘еҷЁbpmn-jsзі»еҲ—д№Ӣеҹәжң¬дҪҝз”ЁгҖҸпјҢ

1.еңЁcomponentsзӣ®еҪ•дёӢж–°е»әcustomBpmnзӣ®еҪ•пјҢеңЁcustomBpmnзӣ®еҪ•дёӢж–°е»әcustomзӣ®еҪ•пјҢжҜҸеұӮзӣ®еҪ•дёӢйғҪж–°е»әindex.jsж–Үд»¶пјҢжңҖз»Ҳзӣ®еҪ•з»“жһ„еҰӮдёӢ

2.еңЁcustomзӣ®еҪ•дёӢж–°е»әCustomPalette.jsж–Үд»¶пјҢеҶ…е®№еҰӮдёӢ

import { assign } from "min-dash";

export default function PaletteProvider(
  palette,
  create,
  elementFactory,
  handTool,
  lassoTool,
  spaceTool,
  globalConnect,
  translate
) {
  this.create = create;
  this.elementFactory = elementFactory;
  this.handTool = handTool;
  this.lassoTool = lassoTool;
  this.spaceTool = spaceTool;
  this.globalConnect = globalConnect;
  this.translate = translate;

  palette.registerProvider(this);
}

PaletteProvider.$inject = [
  "palette",
  "create",
  "elementFactory",
  "handTool",
  "lassoTool",
  "spaceTool",
  "globalConnect",
  "translate"
];

PaletteProvider.prototype.getPaletteEntries = function (element) {
  const {
    create,
    elementFactory,
    handTool,
    lassoTool,
    spaceTool,
    globalConnect,
    translate
  } = this;

  function createAction(type, group, className, title, options) {
    function createListener(event) {
      var shape = elementFactory.createShape(assign({ type: type }, options));

      if (options) {
        shape.businessObject.di.isExpanded = options.isExpanded;
      }

      create.start(event, shape);
    }

    var shortType = type.replace(/^bpmn:/, "");

    return {
      group: group,
      className: className,
      title: title || translate("Create {type}", { type: shortType }),
      action: {
        dragstart: createListener,
        click: createListener
      }
    };
  }

  return {
    "lasso-tool": {
      group: "tools",
      className: "bpmn-icon-lasso-tool",
      title: "Activate the lasso tool",
      action: {
        click: function (event) {
          lassoTool.activateSelection(event);
        }
      }
    },

    "tool-separator": {
      group: "tools",
      separator: true
    },

    "create.start-event": createAction(
      "bpmn:StartEvent",
      "event",
      "bpmn-icon-start-event-none",
      "еҲӣе»әејҖе§ӢиҠӮзӮ№"
    ),
    "create.end-event": createAction(
      "bpmn:EndEvent",
      "event",
      "bpmn-icon-end-event-none",
      "еҲӣе»әз»“жқҹиҠӮзӮ№"
    ),
    "create.user-task": createAction(
      "bpmn:UserTask",
      "activity",
      "bpmn-icon-user-task",
      "еҲӣе»әз”ЁжҲ·д»»еҠЎ"
    ),
    "create.exclusive-gateway": createAction(
      "bpmn:ExclusiveGateway",
      "gateway",
      "bpmn-icon-gateway-xor",
      "еҲӣе»әжҺ’д»–зҪ‘е…і"
    )
  };
};
